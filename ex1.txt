ALGORITHM Dot product 
VAR
    pr  , n INTEGER
    vect1 : ARRAY_OF INTEGER[2];
    vect2: ARRAY_OF INTEGER[2];
    


FUNCTION dot_produit (v1: ARRAY_OF INTEGER[2];, v2 : ARRAY_OF INTEGER[2]; ) : return_type INTEGER
VAR
    ps INTEGER;
BEGIN
    FOR i FROM 0 TO 2 STEP 1  DO
        ps := ps + v1[i]*v2[i];
    END_FOR
    RETURN ps  ;
END
